016public class GameEngine
017{

076    /**
077     * Create all the rooms and link their exits together.
078     */
079    private void createRooms()
080    {
081        Room outside, theatre, pub, lab, office;
082
083        // create the rooms
084        outside = new Room("outside the main entrance of the university", "titre.png");
085        theatre = new Room("in a lecture theatre", "castle.gif");
086        pub = new Room("in the campus pub", "courtyard.gif");
087        lab = new Room("in a computing lab", "stairs.gif");
088        office = new Room("the computing admin office", "dungeon.gif");
089        
090        hmRoom.put("outside", outside);
091        hmRoom.put("theatre", theatre);
092        hmRoom.put("pub", pub);
093        hmRoom.put("lab", lab);
094        hmRoom.put("office", office);
095        
096        // initialise room exits
097        outside.setExit("east", theatre);
098        outside.setExit("south", lab);
099        outside.setExit("west", pub);
100
101        theatre.setExit("west", outside);
102
103        pub.setExit("east", outside);
104
105        lab.setExit("north", outside);
106        lab.setExit("east", office);
107
108        office.setExit("west", lab);
109        
110        //initialise item
111        theatre.setItem("sword", hmItem.get("sword"));
112        theatre.setItem("chair", hmItem.get("chair"));
113        
114        pub.setItem("food", hmItem.get("food"));
115        
116        lab.setItem("keyboard", hmItem.get("keyboard"));
117        
118        office.setItem("card", hmItem.get("card"));
119        
120        currentRoom = outside;  // start game outside
121        lastRoom = outside;
122    }
123
124    /**
125     * Given a command, process (that is: execute) the command.
126     * If this command ends the game, true is returned, otherwise false is
127     * returned.
128     */
129    public void interpretCommand(String commandLine) 
130    {
131        gui.println(commandLine);
132        Command command = parser.getCommand(commandLine);
133
134        if(command.isUnknown()) {
135            gui.println("I don't know what you mean...");
136            return;
137        }
138
139        String commandWord = command.getCommandWord();
140        if (commandWord.equals("help"))
141            printHelp();
142        else if (commandWord.equals("go"))
143            goRoom(command);
144        else if (commandWord.equals("back"))
145            back(command);
146        else if (commandWord.equals("quit")) {
147            if(command.hasSecondWord())
148                gui.println("Quit what?");
149            else
150                endGame();
151        }
152    }

169    /** 
170     * Try to go to one direction. If there is an exit, enter the new
171     * room, otherwise print an error message.
172     */
173    private void goRoom(Command command) 
174    {
175        if(!command.hasSecondWord()) {
176            // if there is no second word, we don't know where to go...
177            gui.println("Go where?");
178            return;
179        }
180
181        String direction = command.getSecondWord();
182
183        // Try to leave current room.
184        Room nextRoom = currentRoom.getExit(direction);
185
186        if (nextRoom == null)
187            gui.println("There is no door!");
188        else {
189            historyRoom.push(currentRoom);
190            lastRoom = currentRoom;
191            currentRoom = nextRoom;
192            gui.println(currentRoom.getLongDescription());
193            gui.println(currentRoom.getItemDescriptionString());
194            if(currentRoom.getImageName() != null)
195                gui.showImage(currentRoom.getImageName());
196        }
197    }
198    
199    private void back(Command command){
200        if(command.hasSecondWord()) {
201            // if there is no second word, we don't know where to go...
202            gui.println("Back What ?");
203            return;
204        }
205        
206        if(historyRoom.empty()){
207            gui.println("UGH UGH UGH Fucking retarded player who want back more what he can.");
208            return;
209        }
210        
211        currentRoom = historyRoom.peek();
212        historyRoom.pop();
213        
214        gui.println(currentRoom.getLongDescription());
215        gui.println(currentRoom.getItemDescriptionString());
216        if(currentRoom.getImageName() != null)
217            gui.showImage(currentRoom.getImageName());
218        else
219            gui.showImage("titre.png");
220    }
221
222    private void endGame()
223    {
224        gui.println("Thank you for playing.  Good bye.");
225        gui.enable(false);
226    }
