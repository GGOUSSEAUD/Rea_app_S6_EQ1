001import java.util.Set;
002import java.util.HashMap;
003import java.util.Iterator;
004
005/*
006 * Class Room - a room in an adventure game.
007 *
008 * This class is part of the "World of Zuul" application. 
009 * "World of Zuul" is a very simple, text based adventure game.  
010 *
011 * A "Room" represents one location in the scenery of the game.  It is 
012 * connected to other rooms via exits.  For each existing exit, the room 
013 * stores a reference to the neighboring room.
014 * 
015 * @author  Michael Kolling and David J. Barnes
016 * @version 1.0 (February 2002) DBMOD:04/04/2008
017 */
018
019public class Room 
020{
021    private String description;
022    private HashMap<String,Room> exits;   
023    private HashMap<String, Item> itemList;// stores exits of this room.
024    private String imageName;
025
026    /**
027     * Create a room described "description" with a given image. 
028     * Initially, it has no exits. "description" is something like 
029     * "in a kitchen" or "in an open court yard".
030     */
031    public Room(String description, String image) 
032    {
033        this.description = description;
034        exits = new HashMap<String,Room>();
035        itemList = new HashMap<String,Item>();
036        imageName = image;
037    }
038
039    /**
040     * Define an exit from this room.
041     */
042    public void setExit(String direction, Room neighbor) 
043    {
044        exits.put(direction, neighbor);
045    }
046    
047    /**
048     * Define an exit from this room.
049     */
050    public void setItem(String itemName, Item item) 
051    {
052        itemList.put(itemName, item);
053    }
054
055    /**
056     * Room's item getter 
057     */
058    public Item getItem(String itemName){
059        return itemList.get(itemName);
060    }
061    
062    /**
063     * Return the description of the room (the one that was defined in the
064     * constructor).
065     */
066    public String getShortDescription()
067    {
068        return description;
069    }
070
071    /**
072     * Return a long description of this room, in the form:
073     *     You are in the kitchen.
074     *     Exits: north west
075     */
076    public String getLongDescription()
077    {
078        return "You are " + description + ".\n" + "Vous trouver :" + getItemString() + "\n" + getExitString();
079    }
080
081    public String getItemString()
082    {
083        StringBuilder returnString = new StringBuilder( "" );
084        if(itemList.isEmpty())
085            returnString.append(" rien.");
086        for ( String vS :itemList.keySet() ){
087            returnString.append(" " + vS);
088        }
089        //returnString.deleteCharAt(returnString.length()-1);
090        return returnString.toString();
091    }
092    
093    public String getItemDescriptionString()
094    {
095        StringBuilder returnString = new StringBuilder( "" );
096        if(!itemList.isEmpty()){
097            for ( String vS :itemList.keySet() ){
098                returnString.append(" " + vS + " : " + itemList.get(vS).getDescription() + "\n");
099            }
100        }
101        return returnString.toString();
102    }
103    
104    /**
105     * Return a string describing the room's exits, for example
106     * "Exits: north west".
107     */
108    public String getExitString()
109    {
110        StringBuilder returnString = new StringBuilder( "Exits:" );
111        for ( String vS : exits.keySet() )
112            returnString.append( " " + vS );
113        return returnString.toString();
114    }
115
116    /**
117     * Return the room that is reached if we go from this room in direction
118     * "direction". If there is no room in that direction, return null.
119     */
120    public Room getExit(String direction) 
121    {
122        return exits.get(direction);
123    }
124
125    /**
126     * Return a string describing the room's image name
127     */
128    public String getImageName()
129    {
130        return imageName;
131    }
132}
